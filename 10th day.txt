   private static int BASE = 2;
    public String addBinary(String a, String b) {
       //string builder to append the intermediate values added 
       StringBuilder sb = new StringBuilder();
        //we start from the least significant bit for addition
        int i = a.length() - 1;
        int j = b.length() - 1;
        //initialize with 0 for both sum and carry
        int sum = 0;
        int carry = 0;
        while (i >= 0 || j >= 0){
            //assign carry to sum as it will be added at each stage
            sum = carry;
            // add the value of the index character to the sum if index is greater than 0
            if (i >= 0) sum += a.charAt(i--)-'0';
            if (j >= 0) sum += b.charAt(j--)-'0';
            //evaluate carry
            carry = sum / BASE;
            //append the mod value with BASE
            sb.append(sum % BASE);
        }
        //if any carry left add to the final result
        if (carry != 0) sb.append(carry);
        //we reverse the string and return
        return sb.reverse().toString();
}
